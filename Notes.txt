Cases:

X1) Camera only
run.py

X2) Camera with model
run.py -m <model>
run.py -m yolov5s

X3) Camera with model and recording
run.py -m <model> -r
run.py -m yolov5s -r



X3) Video only
run.py <videofile>
run.py urban8

X4) Video with model
run.py <-model> <videofile>
run.py -yolov5s urban8

X5) Video with model and logs + recording
run.py <-model> <videofile> -l
run.py -yolov5s urban8 -l -recording

X6) Video with lanes
run.py <videofile> -lanes
run.py urban8 -lanes

X7) Video with model and lanes
run.py <-model> <videofile> -lanes
run.py -yolov5s urban8 -lanes

X8) Video with model and lanes and recording and logs
run.py <-model> <videofile> -l -lanes
OR
run.py <-model> <videofile> -lanes -l

run.py -yolov5s urban8 -l -lanes
